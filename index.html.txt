    // ─── AUTENTICAÇÃO ───
    function autenticar(){
      const exercicio=document.getElementById("login-exercicio").value;
      const perfil=document.getElementById("login-perfil").value;
      const input=document.getElementById("login-input").value;
      
      if(!exercicio){
        showInlineAlert("❌ Selecione um exercício fiscal","error");
        return;
      }
      
      if(!perfil){
        showInlineAlert("❌ Selecione um perfil de acesso","error");
        return;
      }

      let usuario=null;

      if(perfil==="gestor"){
        if(input.trim()!=="123456"){
          showInlineAlert("❌ Senha do Gestor incorreta!\n\nUse: 123456","error");
          return;
        }
        usuario={
          id:"ADM001",
          nome:"Gestor Administrativo",
          perfil:"gestor",
          matricula:"ADM001",
          unidades:["HMWG","HRTM","HEMONORTE","HCCPG"]
        };
      } else if(perfil==="tecnico"){
        if(!input||input.trim()===""){
          showInlineAlert("❌ Digite sua matrícula técnico","error");
          return;
        }
        const tecnicosCadastrados={
          "TEC001":{nome:"Carlos Silva",unidades:["HMWG","HRTM"],servicos:["hemodialise","cirurgias","opme-ortopedia"]},
          "TEC002":{nome:"Ana Costa",unidades:["HRTM"],servicos:["hemodialise","laboratorio"]},
          "TEC003":{nome:"Roberto Martins",unidades:["HMWG","HEMONORTE"],servicos:["telemedicina","homecare"]},
          "TEC004":{nome:"Mariana Santos",unidades:["HCCPG"],servicos:["opme-urologia","cirurgias"]}
        };
        
        const inputUpper=input.toUpperCase().trim();
        if(!tecnicosCadastrados[inputUpper]){
          showInlineAlert("❌ Matrícula não encontrada.\n\nExemplos: TEC001, TEC002, TEC003, TEC004","error");
          return;
        }
        
        const tecnico=tecnicosCadastrados[inputUpper];
        usuario={
          id:inputUpper,
          nome:tecnico.nome,
          perfil:"tecnico",
          matricula:inputUpper,
          unidades:tecnico.unidades,
          servicos:tecnico.servicos
        };
      } else if(perfil==="visitante"){
        usuario={
          id:"VIS001",
          nome:"Visitante",
          perfil:"visitante",
          matricula:"VIS001",
          unidades:[]
        };
      }

      if(usuario){
        usuarioAutenticado=usuario;
        currentProfile=perfil;
        exercicioAtivo=parseInt(exercicio);
        document.getElementById("login-screen").classList.add("hidden");
        document.getElementById("app").classList.remove("hidden");
        document.getElementById("exercicio-badge").textContent=exercicio;

        showInlineAlert("✅ Bem-vindo, "+usuario.nome+"!","info");

    // ─── DASHBOARD ───
    function renderDashboard(el){
      let contratosFiltrados=allContratos.filter(c=>c.anoExercicio==exercicioAtivo);
      if(currentProfile==="tecnico"&&usuarioAutenticado?.unidades){
        contratosFiltrados=contratosFiltrados.filter(c=>
          usuarioAutenticado.unidades.some(u=>c.unidades?.includes(u))
        );
      }
      if(filtroUH) contratosFiltrados=contratosFiltrados.filter(c=>c.unidades?.includes(filtroUH));
      if(filtroServico) contratosFiltrados=contratosFiltrados.filter(c=>c.servicoTipo===filtroServico);
      if(filtroEmpresa) contratosFiltrados=contratosFiltrados.filter(c=>c.empresa.toLowerCase().includes(filtroEmpresa.toLowerCase()));

      const totalContratos=contratosFiltrados.length;
      const totalGlobal=contratosFiltrados.reduce((s,c)=>s+(parseFloat(c.valorContrato)||0),0);
      const totalEmpenhado=contratosFiltrados.reduce((s,c)=>s+(parseFloat(c.valorEmpenho2026)||0),0);
      const totalExecutado=contratosFiltrados.reduce((s,c)=>s+(parseFloat(c.valorExecutado2026)||0),0);
      const saldo=totalEmpenhado-totalExecutado;
      
      const cards=[
        {icon:"landmark",label:"Valor Global",value:fmtM(totalGlobal),sub:"contratado",color:"navy"},
        {icon:"trending-up",label:"Total Empenhado",value:fmtM(totalEmpenhado),sub:pct(totalEmpenhado,totalGlobal)+"%",color:"blue"},
        {icon:"wallet",label:"Valor Executado",value:fmtM(totalExecutado),sub:pct(totalExecutado,totalEmpenhado)+"%",color:"emerald"},
        {icon:"zap",label:"Saldo Disponível",value:fmtM(saldo),sub:pct(saldo,totalEmpenhado)+"%",color:"cyan"},
      ];

        <div class="glass rounded-xl p-6 flex flex-wrap gap-4 items-center">

              <option value="2025" ${exercicioAtivo===2025?"selected":""}>2025</option>
              <option value="2024" ${exercicioAtivo===2024?"selected":""}>2024</option>
            </select>

            <label class="block text-sm font-bold text-navy-900 mb-2">Unidade Hospitalar</label>

    async function salvarContrato(e){
      e.preventDefault();
      if(!hasPermission("create")){showInlineAlert("Sem permissão","error");return}
      
      const unidades=[...document.querySelectorAll(".form-uh-multi:checked")].map(el=>el.value);
      if(unidades.length===0){
        showInlineAlert("Selecione pelo menos uma unidade hospitalar","error");
        return;
      }
      
      const novoContrato={
        id:Date.now().toString(),
        tipo:"contrato",
        nProcesso:document.getElementById("form-nprocesso").value,
        unidades:unidades.join(","),
        empresa:document.getElementById("form-empresa").value,
        nContrato:document.getElementById("form-ncontrato").value,
        aditivo:document.getElementById("form-aditivo").value||"—",
        vigIni:document.getElementById("form-vigini").value,
        vigFim:document.getElementById("form-vigfim").value,
        valorContrato:parseFloat(document.getElementById("form-valor-contrato").value),
        valorEstimadoMensal:parseFloat(document.getElementById("form-valor-mensal").value),
        servicoTipo:document.getElementById("form-servico").value,
        anoExercicio:exercicioAtivo,
        status:"Ativo",
        valorEmpenho2026:0,
        valorExecutado2026:0,
        criadoEm:new Date().toISOString()
      };
      
      const result=await window.dataSdk.create(novoContrato);
      if(result.isOk){
        closeAllModals();
        showInlineAlert("✅ Contrato salvo com sucesso!","info");
      } else {
        showInlineAlert("Erro ao salvar contrato","error");
      }
    }

    function adicionarItem(tipo){
      const inputs={
        empenho:"input-empenho",
        fonte:"input-fonte",
        unidade:"input-unidade",
        empresa:"input-empresa",
        situacao:"input-situacao",
        status:"input-status",
        servidor:"input-servidor"
      };
      
      const inputEl=document.getElementById(inputs[tipo]);
      if(!inputEl||!inputEl.value.trim()){
        showInlineAlert("❌ Digite um valor","error");
        return;
      }
      
      showInlineAlert("✅ Item adicionado com sucesso!","info");
      inputEl.value="";
      renderConfiguracoes(document.getElementById("content-area"));
    }

    async function initializeApp(){

          console.error("Data SDK error:",e);
        }
      }

          window.elementSdk.init({

              const titleEl=document.getElementById("main-title");

              if(subtitleEl) subtitleEl.textContent=cfg.subtitle_text;

            mapToCapabilities:cfg=>({recolorables:[],borderables:[],fontEditable:undefined,fontSizeable:undefined}),
            mapToEditPanelValues:cfg=>new Map([["system_title",cfg.system_title],["subtitle_text",cfg.subtitle_text]])
          });
        } catch(e){
          console.error("Element SDK error:",e);
        }
      }
      
      lucide.createIcons();
    }
